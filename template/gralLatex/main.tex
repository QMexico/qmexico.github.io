%% As an option to frame
\defbeamertemplate*{background canvas}{mydefault}{%
  \ifbeamercolorempty[bg]{background canvas}{}{\color{bg}\vrule width\paperwidth height\paperheight}}
\defbeamertemplate*{background canvas}{image}
{%
    \begin{tikzpicture}
      \useasboundingbox (0,0) rectangle(
      \the\paperwidth, \the\paperheight);
      % {cx}{cy} cambia la siguiente linea en los valores cx y cy
        \pgftext[at=\pgfpoint{0cm}{0cm}, left, base]{
          \pgfsetfillopacity{0.1}\pgfuseimage{mybackground}
        }; 
    \end{tikzpicture}
}
\BeforeBeginEnvironment{frame}{%
  \setbeamertemplate{background canvas}[mydefault]%
}
